
# 전력망을 둘로 나누기
  - [전력망을 둘로 나누기](https://programmers.co.kr/learn/courses/30/lessons/86971)


### solution
  - 연결을 하나씩 모두 끊으며 양쪽 트리 개수 비교
  - 중간 위치의 노드를 찾을 필요 없이 모든 경우를 탐색

  ```javascript
  function createVisited(n){
    return new Array(n+1).fill(0).map(_=>false)
  }

  function bfs(graph, node, visited){
      let count = 0;
      const q = [];
      q.push(node);
      visited[node] = true;
      while(q.length>0){
          const current = q.shift();
          const nextNodes = graph[current];
          for(let next of nextNodes){
              if(visited[next]) continue;
              visited[next] = true;
              q.push(next);
          }
          count += 1;
      }
      return count
  }

  function solution(n, wires) {
      let answer = Infinity;
      const graph = new Array(n+1).fill(0).map(_=>[])
      wires.forEach(arr=>{
          graph[arr[0]].push(arr[1]);
          graph[arr[1]].push(arr[0]);
      })
      for(let i=1; i<n+1; i++){
          for(let j of graph[i]){
              const copy = graph.map(arr=>[...arr]);
              copy[i].splice(copy[i].indexOf(j),1);
              copy[j].splice(copy[j].indexOf(i),1);
              answer = Math.min(answer, Math.abs(
                  bfs(copy, i, createVisited(n))-bfs(copy, j, createVisited(n))
              ));
          }
      }
      return answer
  }
  ```
