### 풀이
  - 특정 문자열을 원하는 방법대로 분할하는 능력을 확인하는 문제

```javascript
function solution(s) {
    let answer = [];    
    let str = s.slice(2,s.length-2).split("},{").map(arr=> arr.split(','))
    str.sort((x,y) => x.length-y.length).forEach(arr=>{
        for (let num of arr){
            if(!answer.includes(num)) answer.push(num)
        }
    })
    
    return answer.map(Number);
}
```
