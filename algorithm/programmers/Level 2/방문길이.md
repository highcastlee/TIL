
# 방문길이
 - [방문길이](https://programmers.co.kr/learn/courses/30/lessons/49994#)


### solution
  - 좌표 전후 해시 키로 양방향 기록
  - 해시에 없는 길만 answer += 1

  ```javascript
  function solution(dirs) {
    let answer = 0;
    let before = [0,0]
    const current = [0,0]
    const hash = {}
    for(let dir of [...dirs]){
        if(dir === "U") {
            if(current[0]-1 < -5) continue
            current[0] -= 1
        }
        if(dir === "D"){
            if(current[0]+1 > 5) continue
            current[0] += 1
        }
        if(dir === "L"){
            if(current[1]-1 < -5) continue
            current[1] -= 1
        }
        if(dir === "R"){
            if(current[1]+1 > 5) continue
            current[1] += 1
        }
        let key1 = before.join('')+current.join('')
        let key2 = current.join('')+before.join('')
        if(!hash[key1] && !hash[key2]) {
            hash[key1] = true
            hash[key2] = true
            answer += 1
        }
        before = [...current]
    }
    return answer;
  }
  ```
