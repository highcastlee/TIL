# 키패드 누르기
 - [키패드 누르기](https://programmers.co.kr/learn/courses/30/lessons/67256)


### 풀이
  - 키패드 그래프 만들어서 row,col 거리 계산하여 left or right 선택
  - 선택될 때마다 leftHand, rightHand의 위치를 알맞게 갱신한다.


  ```javascript
  function solution(numbers, hand) {
      const keyPad = {
          1: [0,0],
          2: [0,1],
          3: [0,2],
          4: [1,0],
          5: [1,1],
          6: [1,2],
          7: [2,0],
          8: [2,1],
          9: [2,2],
          0: [3,1],
      };
      const leftNums = [1,4,7];
      const rightNums = [3,6,9];
      let leftHand = {row:3, col:0};
      let rightHand = {row:3, col:2};

      const result = [];

      for(const currentNum of numbers){
          const [currentRow,currentCol] = keyPad[currentNum];
          const currentObj = {row:currentRow,col:currentCol}
          if(leftNums.includes(currentNum)){
              result.push('L');
              leftHand = currentObj;
          }else if(rightNums.includes(currentNum)){
              result.push('R');
              rightHand = currentObj;
          }else{
              let distanceR = Math.abs(rightHand.row-currentRow)+Math.abs(rightHand.col-currentCol);
              let distanceL = Math.abs(leftHand.row-currentRow)+Math.abs(leftHand.col-currentCol);
              if(distanceL === distanceR){
                  result.push(hand==='right' ? 'R' : 'L');
                  (hand === 'right') 
                      ? rightHand = currentObj 
                      : leftHand = currentObj;
              }else if(distanceL < distanceR){
                  leftHand = currentObj;
                  result.push('L');
              }else{
                  rightHand = currentObj;
                  result.push('R');                
              }
          }
      }
      return result.join("");
  }
  ```
